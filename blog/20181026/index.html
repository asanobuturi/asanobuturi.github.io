#include "DxLib.h"

int GetFps()
{
	static int OldTime = -1;
	static int Count = 0;

	static int result = -1;

	if (OldTime == -1) {
		OldTime = GetNowCount();
		Count = -1;
	}
	Count++;

	if (Count == 10) {
		result = (int)(1000.*10. / (GetNowCount() - OldTime) + 0.5);
		OldTime = GetNowCount();
		Count = 0;
	}

	return result;
}

int WINAPI WinMain(HINSTANCE hInstance, HINSTANCE hPrevInstance,
	LPSTR lpCmdLine, int nCmdShow)
{
	int WindowSizeX,WindowSizeY,WindowColorBit;

	GetDefaultState(&WindowSizeX, &WindowSizeY, &WindowColorBit);
	SetGraphMode(WindowSizeX, WindowSizeY, WindowColorBit);

	//ChangeWindowMode( TRUE );

	SetBackgroundColor(255, 255, 255);

	if (DxLib_Init() == -1)
		return -1;

	SetDrawScreen(DX_SCREEN_BACK);

	while (ProcessMessage() == 0 && CheckHitKey(KEY_INPUT_ESCAPE) == 0) {
		ClearDrawScreen();



		ScreenFlip();
	}

	DxLib_End();
	return 0;
}
